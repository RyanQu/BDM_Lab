The query is
SELECT DISTINCT P.ticker
FROM price P
WHERE P.close > (
    SELECT DISTINCT P_1.close 
    FROM price P_1 
    WHERE P_1.date='2017-03-20' 
    AND P_1.ticker='IBM')
AND P.close <= (
    SELECT DISTINCT P_2.close 
    FROM price P_2 
    WHERE P_2.date='2017-03-20' 
    AND P_2.ticker='GOOG');
